CREATE DATABASE fakultet;

CREATE TABLE studenti (
    student_id INT PRIMARY KEY,
    ime VARCHAR(50),
    prezime VARCHAR(50),
    godina_upisa INT
);

CREATE TABLE predmeti (
    predmet_id INT PRIMARY KEY,
    naziv_predmeta VARCHAR(100),
    espb INT
);

CREATE TABLE ispiti (
    ispit_id INT PRIMARY KEY,
    student_id INT,
    predmet_id INT,
    datum_polaganja DATE,
    ocena INT,
    FOREIGN KEY (student_id) REFERENCES studenti(student_id),
    FOREIGN KEY (predmet_id) REFERENCES predmeti(predmet_id)
);

CREATE TABLE ispiti (
    ispit_id INT PRIMARY KEY,
    student_id INT,
    predmet_id INT,
    datum_polaganja DATE,
    ocena INT,
    FOREIGN KEY (student_id) REFERENCES studenti(student_id),
    FOREIGN KEY (predmet_id) REFERENCES predmeti(predmet_id)
);

-- INNER JOIN – Prikazuje samo studente koji su polagali ispite.
CREATE VIEW view_student_ispiti AS
SELECT 
    s.student_id,
    s.ime,
    s.prezime,
    p.naziv_predmeta,
    i.datum_polaganja,
    i.ocena
FROM studenti s
INNER JOIN ispiti i ON s.student_id = i.student_id
INNER JOIN predmeti p ON i.predmet_id = p.predmet_id;

--  JOIN – Prikazuje sve studente, bez obzira da li su polagali ispite ili ne.
CREATE VIEW view_student_ispiti_left AS
SELECT 
    s.student_id,
    s.ime,
    s.prezime,
    p.naziv_predmeta,
    i.datum_polaganja,
    i.ocena
FROM studenti s
LEFT JOIN ispiti i ON s.student_id = i.student_id
LEFT JOIN predmeti p ON i.predmet_id = p.predmet_id;
